function isNumber(obj){
    var reg = new RegExp("^[1-9][0-9]*$");
    return reg.test(obj);
}

function isTowScaleNumber(obj) {
    if(obj == null || obj == ""){
        return true;
    }
    var reg = new RegExp("^[0-9]+(\.[0-9]{0,2})?$");
    return reg.test(obj);
}
